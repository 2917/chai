TDD Style Introduction

module.exports()

The TDD style is exposed through assert interfaces. This provides the classic assert.test notation, similiar to that packaged with node.js. This assert module, however, provides several additional tests and is browser compatible. 

 // assert
 var assert = require('chai').assert;
   , foo = 'bar';

 assert.typeOf(foo, 'string');
 assert.equal(foo, 'bar');

Configuration

By default, Chai does not show stack traces upon an AssertionError. This can be changed by modifying the includeStack parameter for chai.Assertion. For example: 

 var chai = require('chai');
 chai.Assertion.includeStack = true; // defaults to false

.include(value, includes, [message])

@param{ * }includes
@param{ String }message
@apipublic

Assert the inclusion of an object in another. Works for strings and arrays. 

 assert.include('foobar', 'bar', 'foobar contains string `var`);

View Source
assert.include = function (exp, inc, msg) {
    var obj = new Assertion(exp, msg);

    if (Array.isArray(exp)) {
      obj.to.include(inc);
    } else if ('string' === typeof exp) {
      obj.to.contain.string(inc);
    }
  };

assert.operator = function (val, operator, val2, msg) {
    if (!~['==', '===', '>', '>=', '<', '<=', '!=', '!=='].indexOf(operator)) {
      throw new Error('Invalid operator "' + operator + '"');
    }
    var test = new Assertion(eval(val + operator + val2), msg);
    test.assert(
        true === flag(test, 'object')
      , 'expected ' + util.inspect(val) + ' to be ' + operator + ' ' + util.inspect(val2)
      , 'expected ' + util.inspect(val) + ' to not be ' + operator + ' ' + util.inspect(val2) );
  };
